Button with Data Attributes

Java

public record DataButton(
    String id,
    String name,
    String value,
    String text,
    String type,
    String variant,
    String action,
    String confirm,
    String ariaLabel) {

  public DataButton() {
    this("deleteBtn",
         "action",
         "delete",
         "Delete",
         "submit",
         "base",
         "delete-record",
         "Are you sure you want to delete this?",
         "Delete this record");
  }
}

Thymeleaf

<cf:button
  th:id="${model.dataButton.id()}"
  th:name="${model.dataButton.name()}"
  th:value="${model.dataButton.value()}"
  th:text="${model.dataButton.text()}"
  th:type="${model.dataButton.type()}"
  th:variant="${model.dataButton.variant()}"
  th:data-action="${model.dataButton.action()}"
  th:data-confirm="${model.dataButton.confirm()}"
  th:aria-label="${model.dataButton.ariaLabel()}"
/>

Rendered HTML

<button
  type="submit"
  class="usa-button usa-button--base"
  id="deleteBtn"
  name="action"
  value="delete"
  data-action="delete-record"
  data-confirm="Are you sure you want to delete this?"
  aria-label="Delete this record">Delete</button>

Button with Icon (Nested Content)

Java

public record IconButton(
    String id,
    String variant,
    String ariaLabel,
    String text) {

  public IconButton() {
    this("menuBtn",
         "accent-warm",
         "Delete",
         "Delete");
  }
}

Thymeleaf

<cf:button
  th:id="${model.iconButton.id()}"
  th:variant="${model.iconButton.variant()}"
  th:aria-label="${model.iconButton.ariaLabel()}"
>
  <cf:icon type="icon-delete" />
  <span th:text="${model.iconButton.text()}"></span>
</cf:button>

Rendered HTML

<button
  type="button"
  class="usa-button usa-button--accent-warm"
  id="menuBtn"
  aria-label="Delete">
  <img
    src="/assets/Images/uswds/icon-delete.svg"
    aria-hidden="true"
    role="img"
    alt="">
  <span>Delete</span>
</button>
