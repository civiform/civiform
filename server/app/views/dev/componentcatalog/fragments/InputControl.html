Input Component

Parameter Name Required Description
id Yes Unique identifier for the input element
name Yes The name attribute for form submission
disabled No Disable the input (defaults to "false")
help-text No Optional help text to display
is-valid No Controls validation state (defaults to "true")
label No The label text to display for the input
max No Maximum value for number/date inputs
maxlength No Maximum length for text inputs
min No Minimum value for number/date inputs
minlength No Minimum length for text inputs
pattern No Regular expression pattern for validation
placeholder No Placeholder text for the input
readonly No Make the input readonly (defaults to "false")
required No Makes the input required
size No Size of the input element (affects CSS class)
type No Input type (defaults to "text")
validation-class No CSS class for validation styling
validation-field No Field name for validation
validation-message No Custom validation error message
value No The value of the input element

Basic Text Input

Code Sample

Java

public class BasicTextInput {
      public String id() {
        return "firstName";
      }

      public String name() {
        return "firstName";
      }

      public String label() {
        return "First Name";
      }

      public String placeholder() {
        return "Enter your first name";
      }
    }

Thymeleaf

<cf:input
      th:id="${model.basicInput.id()}"
      th:name="${model.basicInput.name()}"
      th:label="${model.basicInput.label()}"
      th:placeholder="${model.basicInput.placeholder()}"
    />

Rendered HTML

<div class="usa-form-group">
      <label class="usa-label" for="firstName">First Name</label>
      <input
        type="text"
        class="usa-input"
        id="firstName"
        name="firstName"
        placeholder="Enter your first name" />
    </div>

Password Input with Validation Error

Code Sample

Java

public class PasswordInput {
      public String id() {
        return "userPassword";
      }

      public String name() {
        return "password";
      }

      public String label() {
        return "Password";
      }

      public String type() {
        return "password";
      }

      public String isValid() {
        return "false";
      }

      public String validationMessage() {
        return "Password must be at least 8 characters";
      }
    }

Thymeleaf

<cf:input
      th:id="${model.passwordInput.id()}"
      th:name="${model.passwordInput.name()}"
      th:label="${model.passwordInput.label()}"
      th:type="${model.passwordInput.type()}"
      th:is-valid="${model.passwordInput.isValid()}"
      th:validation-message="${model.passwordInput.validationMessage()}"
      required
    />

Rendered HTML

<div class="usa-form-group usa-form-group--error">
      <label class="usa-label" for="userPassword">Password</label>
      <span id="userPassword-error" class="usa-error-message" role="alert">
        Password must be at least 8 characters
      </span>
      <input
        type="password"
        class="usa-input usa-input--error"
        id="userPassword"
        name="password"
        required="required"
        aria-describedby="userPassword-error"
        aria-invalid="true" />
    </div>

Input with Data Attributes

Code Sample

Java

public class DataAttributeInput {
      public String id() {
        return "searchQuery";
      }

      public String name() {
        return "q";
      }

      public String label() {
        return "Search";
      }

      public String testId() {
        return "search-input";
      }

      public String analyticsEvent() {
        return "search-initiated";
      }

      public String ariaLabel() {
        return "Search our website";
      }
    }

Thymeleaf

<cf:input
      th:id="${model.dataInput.id()}"
      th:name="${model.dataInput.name()}"
      th:label="${model.dataInput.label()}"
      th:data-test-id="${model.dataInput.testId()}"
      th:data-analytics="${model.dataInput.analyticsEvent()}"
      th:aria-label="${model.dataInput.ariaLabel()}"
    />

Rendered HTML

<div class="usa-form-group">
      <label class="usa-label" for="searchQuery">Search</label>
      <input
        type="text"
        class="usa-input"
        id="searchQuery"
        name="q"
        data-test-id="search-input"
        data-analytics="search-initiated"
        aria-label="Search our website" />
    </div>
