Select Component

Basic Select

Java

import com.google.common.collect.ImmutableList;

public record BasicSelect(
    String id,
    String name,
    String label,
    ImmutableList<Option> options,
    String selectedValue) {

  public static BasicSelect create() {
    return new BasicSelect(
        "color",
        "color",
        "Favorite Color",
        ImmutableList.of(
            new Option("red", "Red"),
            new Option("blue", "Blue"),
            new Option("green", "Green")),
        "blue");
  }
}

public record Option(String value, String text) {}

Thymeleaf

<cf:select
  th:id="${model.basicSelect.id()}"
  th:name="${model.basicSelect.name()}"
  th:label="${model.basicSelect.label()}"
>
  <option
    th:each="item: ${model.basicSelect.options()}"
    th:value="${item.value()}"
    th:text="${item.text()}"
    th:selected="${item.value() == model.basicSelect.selectedValue()}"
  ></option>
</cf:select>

Rendered HTML

<div class="usa-form-group">
  <label for="color" class="usa-label">Favorite Color</label>
  <select class="usa-select" id="color" name="color">
    <option value="red">Red</option>
    <option value="blue" selected>Blue</option>
    <option value="green">Green</option>
  </select>
</div>

Select with Validation Error

Java

import com.google.common.collect.ImmutableList;

public record SelectWithError(
    String id,
    String name,
    String label,
    String isValid,
    String validationMessage,
    ImmutableList<Option> options) {

  public static SelectWithError create() {
    return new SelectWithError(
        "state",
        "state",
        "State",
        "false",
        "Please select a state",
        ImmutableList.of(
            new Option("ca", "California"),
            new Option("ny", "New York"),
            new Option("tx", "Texas")));
  }
}

public record Option(String value, String text) {}

Thymeleaf

<cf:select
  th:id="${model.selectWithError.id()}"
  th:name="${model.selectWithError.name()}"
  th:label="${model.selectWithError.label()}"
  th:is-valid="${model.selectWithError.isValid()}"
  th:validation-message="${model.selectWithError.validationMessage()}"
  required
>
  <option value="">-- Please select --</option>
  <option
    th:each="item: ${model.selectWithError.options()}"
    th:value="${item.value()}"
    th:text="${item.text()}"
  ></option>
</cf:select>

Rendered HTML

<div class="usa-form-group usa-form-group--error">
  <label for="state" class="usa-label">State</label>
  <select
    class="usa-select usa-input--error"
    id="state"
    name="state"
    required="required"
    aria-describedby="state-error"
    aria-invalid="true">
    <option value="">-- Please select --</option>
    <option value="ca">California</option>
    <option value="ny">New York</option>
    <option value="tx">Texas</option>
  </select>
  <span id="state-error" class="usa-error-message" role="alert">
    Please select a state
  </span>
</div>

Select with Option Groups

Java

import com.google.common.collect.ImmutableList;

public record GroupedSelect(
    String id,
    String name,
    String label,
    ImmutableList<Option> fruits,
    ImmutableList<Option> vegetables) {

  public static GroupedSelect create() {
    return new GroupedSelect(
        "food",
        "food",
        "Choose a Food",
        ImmutableList.of(
            new Option("apple", "Apple"),
            new Option("banana", "Banana")),
        ImmutableList.of(
            new Option("carrot", "Carrot"),
            new Option("broccoli", "Broccoli")));
  }
}

public record Option(String value, String text) {}

Thymeleaf

<cf:select
  th:id="${model.groupedSelect.id()}"
  th:name="${model.groupedSelect.name()}"
  th:label="${model.groupedSelect.label()}"
>
  <optgroup label="Fruits">
    <option
      th:each="item: ${model.groupedSelect.fruits()}"
      th:value="${item.value()}"
      th:text="${item.text()}"
    ></option>
  </optgroup>
  <optgroup label="Vegetables">
    <option
      th:each="item: ${model.groupedSelect.vegetables()}"
      th:value="${item.value()}"
      th:text="${item.text()}"
    ></option>
  </optgroup>
</cf:select>

Rendered HTML

<div class="usa-form-group">
  <label for="food" class="usa-label">Choose a Food</label>
  <select class="usa-select" id="food" name="food">
    <optgroup label="Fruits">
      <option value="apple">Apple</option>

    </optgroup>
    <optgroup label="Vegetables">
      <option value="carrot">Carrot</option>

import com.google.common.collect.ImmutableList;

public record SelectWithData(
    String id,
    String name,
    String label,
    String testId,
    String category,
    ImmutableList<Option> options,
    String selectedValue) {

  public static SelectWithData create() {
    return new SelectWithData(
        "size",
        "size",
        "T-Shirt Size",
        "size-selector",
        "clothing",
        ImmutableList.of(
            new Option("s", "Small"),
            new Option("m", "Medium"),
            new Option("l", "Large")),
        "m");
  }
}

public record Option(String value, String text) {}

Thymeleaf

<cf:select
  th:id="${model.selectWithData.id()}"

  th:data-category="${model.selectWithData.category()}"
>
  <option
    th:each="item: ${model.selectWithData.options()}"
    th:value="${item.value()}"
    th:text="${item.text()}"
    th:selected="${item.value() == model.selectWithData.selectedValue()}"
  ></option>
</cf:select>

Rendered HTML

<div class="usa-form-group">
