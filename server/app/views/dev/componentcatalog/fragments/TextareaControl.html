Textarea Component

Basic Textarea

Java

public record BasicTextarea(
    String id,
    String name,
    String label,
    String placeholder) {

  public BasicTextarea() {
    this("comments",
         "comments",
         "Comments",
         "Enter your comments here");
  }
}

Thymeleaf

<cf:textarea
  th:id="${model.basicTextarea.id()}"
  th:name="${model.basicTextarea.name()}"
  th:label="${model.basicTextarea.label()}"
  th:placeholder="${model.basicTextarea.placeholder()}"
/>

Rendered HTML

<div class="usa-form-group">
  <label class="usa-label" for="comments">Comments</label>
  <textarea
    class="usa-textarea"
    id="comments"
    name="comments"
    placeholder="Enter your comments here"></textarea>
</div>

public record TextareaWithHelp(

  public TextareaWithHelp() {

Textarea with Validation Error

Java

public record TextareaWithError(
    String id,
    String name,
    String label,
    String isValid,
    String validationMessage,
    String rows) {

  public TextareaWithError() {
    this("feedback",
         "feedback",
         "Feedback",
         "false",
         "Feedback must be at least 10 characters",
         "4");
  }
}

Thymeleaf

<cf:textarea
  th:id="${model.textareaWithError.id()}"
  th:name="${model.textareaWithError.name()}"
  th:label="${model.textareaWithError.label()}"
  th:is-valid="${model.textareaWithError.isValid()}"
  th:validation-message="${model.textareaWithError.validationMessage()}"
  th:rows="${model.textareaWithError.rows()}"
  required
/>

Rendered HTML

<div class="usa-form-group usa-form-group--error">
  <label class="usa-label" for="feedback">Feedback</label>
  <span id="feedback-error" class="usa-error-message" role="alert">
    Feedback must be at least 10 characters
  </span>
  <textarea
    class="usa-textarea usa-input--error"
    id="feedback"
    name="feedback"
    rows="4"
    required="required"
    aria-describedby="feedback-error"
    aria-invalid="true"></textarea>
</div>

  public TextareaWithValue() {

Textarea with Data Attributes

Java

public record TextareaWithData(
    String id,
    String name,
    String label,
    String testId,
    String maxLength,
    String ariaLabel,
    String rows) {

  public TextareaWithData() {
    this("message",
         "message",
         "Message",
         "message-textarea",
         "500",
         "Enter your message here",
         "8");
  }
}

Thymeleaf

<cf:textarea
  th:id="${model.textareaWithData.id()}"
  th:name="${model.textareaWithData.name()}"
  th:label="${model.textareaWithData.label()}"
  th:data-test-id="${model.textareaWithData.testId()}"
  th:data-max-length="${model.textareaWithData.maxLength()}"
  th:aria-label="${model.textareaWithData.ariaLabel()}"
  th:rows="${model.textareaWithData.rows()}"
  required
/>

Rendered HTML

<div class="usa-form-group">
  <label class="usa-label" for="message">Message</label>
  <textarea
    class="usa-textarea"
    id="message"
    name="message"
    rows="8"
    required="required"
    data-test-id="message-textarea"
    data-max-length="500"
    aria-label="Enter your message here"></textarea>
</div>
