Basic Radio Group

Java

public record BasicRadio(
    String name,
    String label,
    ImmutableList<RadioOption> options) {

  public BasicRadio() {
    this("size",
         "Choose a size",
         ImmutableList.of(
             new RadioOption("size-small", "small", "Small"),
             new RadioOption("size-medium", "medium", "Medium"),
             new RadioOption("size-large", "large", "Large")));
  }
}

public record RadioOption(String id, String value, String label) {}

Thymeleaf

<cf:radio
  th:name="${model.basicRadio.name()}"
  th:label="${model.basicRadio.label()}"
>
  <item
    th:each="item: ${model.basicRadio.options()}"
    th:id="${item.id()}"
    th:value="${item.value()}"
    th:label="${item.label()}"
  ></item>
</cf:radio>

public record RadioWithError(

    String isValid,
    String validationMessage,
    ImmutableList<RadioOption> options) {

  public RadioWithError() {
    this("paymentMethod",
         "Payment Method",
         "false",

         ImmutableList.of(
             new RadioOption("payment-card", "card", "Credit Card"),
             new RadioOption("payment-bank", "bank", "Bank Transfer"),
             new RadioOption("payment-paypal", "paypal", "PayPal")));
  }
}

public record RadioOption(String id, String value, String label) {}

  th:name="${model.radioWithError.name()}"

  <item
    th:each="item: ${model.radioWithError.options()}"

public record RadioNonTiled(
    String name,
    String label,
    String tiled,
    ImmutableList<RadioOptionWithSelected> options) {

  public RadioNonTiled() {
    this("delivery",
         "Delivery Speed",
         "false",
         ImmutableList.of(
             new RadioOptionWithSelected(
                 "delivery-standard", "standard", "Standard (5-7 days)", true),
             new RadioOptionWithSelected(
                 "delivery-express", "express", "Express (2-3 days)", false),
             new RadioOptionWithSelected(
                 "delivery-overnight", "overnight", "Overnight", false)));
  }
}

public record RadioOptionWithSelected(
    String id, String value, String label, boolean selected) {}

<cf:radio
