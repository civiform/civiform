Checkbox Component

Basic Checkbox Group

Code sample

Java

import com.google.common.collect.ImmutableList;

public class BasicCheckbox {
  public record Option(String id, String value, String label) {}

  public String name() {
    return "interests";
  }

  public String label() {
    return "Select your interests";
  }

  public ImmutableList<Option> items() {
    return ImmutableList.of(
        new Option("interest-reading", "reading", "Reading"),
        new Option("interest-sports", "sports", "Sports"),
        new Option("interest-music", "music", "Music"));
  }
}

Thymeleaf

<cf:checkbox
  th:name="${model.basicCheckbox.name()}"
  th:label="${model.basicCheckbox.label()}"
>
  <item
    th:each="item: ${model.basicCheckbox.items()}"
    th:id="${item.id()}"
    th:value="${item.value()}"
    th:label="${item.label()}"
  ></item>
</cf:checkbox>

Checkbox with Validation Error

Code sample

Java

import com.google.common.collect.ImmutableList;

public class CheckboxWithError {
  public record Option(String id, String value, String label) {}

  public String name() {
    return "terms";
  }

  public String label() {
    return "Terms and conditions";
  }

  public String isValid() {
    return "false";
  }

  public String validationMessage() {
    return "You must accept at least one term to continue";
  }

  public ImmutableList<Option> items() {
    return ImmutableList.of(
        new Option("terms-service", "service", "I agree to the Terms of Service"),
        new Option("terms-privacy", "privacy", "I agree to the Privacy Policy"));
  }
}

Thymeleaf

<cf:checkbox
  th:name="${model.checkboxWithError.name()}"
  th:label="${model.checkboxWithError.label()}"
  th:is-valid="${model.checkboxWithError.isValid()}"
  th:validation-message="${model.checkboxWithError.validationMessage()}"
  required
>
  <item
    th:each="item: ${model.checkboxWithError.items()}"
    th:id="${item.id()}"
    th:value="${item.value()}"
    th:label="${item.label()}"
  ></item>
</cf:checkbox>

Rendered HTML

<div class="usa-form-group usa-form-group--error">
  <fieldset class="usa-fieldset">

Checkbox with Pre-selected Items (Non-tiled)

Code sample

Java

import com.google.common.collect.ImmutableList;

public class CheckboxSelected {
  public record Option(
      String id,
      String value,
      String label,
      boolean selected) {}

  public String name() {
    return "skills";
  }

  public String label() {
    return "Programming skills";
  }

  public String tiled() {
    return "false";
  }

  public ImmutableList<Option> items() {
    return ImmutableList.of(
        new Option("skill-java", "java", "Java", true),
        new Option("skill-python", "python", "Python", true),
        new Option("skill-javascript", "javascript", "JavaScript", false),
        new Option("skill-csharp", "csharp", "C#", false));
  }
}

Thymeleaf

<cf:checkbox
  th:name="${model.checkboxSelected.name()}"
  th:label="${model.checkboxSelected.label()}"
  th:tiled="${model.checkboxSelected.tiled()}"

    th:id="${item.id()}"
    th:value="${item.value()}"
    th:label="${item.label()}"
    th:selected="${item.selected()}"
  ></item>
</cf:checkbox>

Checkbox with Data Attributes

Code sample

Java

import com.google.common.collect.ImmutableList;

public class CheckboxWithData {
  public record Option(
      String id,
      String value,
      String label,
      String testId,
      String analytics,
      boolean disabled) {}

  public String name() {
    return "features";
  }

  public String label() {
    return "Available features";
  }

  public ImmutableList<Option> items() {
    return ImmutableList.of(
        new Option(
            "feature-basic",
            "basic",
            "Basic features",
            "checkbox-basic",
            "feature-basic-selected",
            false),
        new Option(
            "feature-premium",
            "premium",
            "Premium features",
            "checkbox-premium",
            "feature-premium-selected",
            true));
  }
}

Thymeleaf

<cf:checkbox
  th:name="${model.checkboxWithData.name()}"
  th:label="${model.checkboxWithData.label()}"
>
  <item
    th:each="item: ${model.checkboxWithData.items()}"
    th:id="${item.id()}"
    th:value="${item.value()}"
    th:label="${item.label()}"
    th:data-test-id="${item.testId()}"
    th:data-analytics="${item.analytics()}"
    th:disabled="${item.disabled()}"
  ></item>
</cf:checkbox>

Rendered HTML

    </div>
