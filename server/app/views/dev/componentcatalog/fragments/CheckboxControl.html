Name Required Type Details
Attributes
id required string Unique identifier for the checkbox group.
label required string Legend text for the checkbox group fieldset.
name required string Form field name shared by all checkboxes in the group.
aria-* optional string Any aria-* attributes on the group are passed through to rendered elements.
columns optional integer string Number of columns to arrange the checkboxes. Allowed options: 2, 3, 4
data-* optional string Any data-* attributes on the group are passed through to rendered elements.
disabled optional boolean Set to "true" to disable all checkboxes in the group.
help-text optional string Helper text displayed below the legend.
is-valid optional boolean Controls validation state. "true" or "false". Defaults to "true".
readonly optional boolean Set to "true" to make all checkboxes in the group read-only.
required optional boolean Set to "true" to mark the group as required.
tiled optional boolean Set to "true" or "false" to use USWDS tile styling. Defaults to "true".
validation-class optional string Fully qualified Java class name used to derive client-side validation constraints.
validation-field optional string Field name within the validation class to extract Play Framework validation constraints from.
validation-message optional string Error message displayed when the group is invalid.
Nested content (<item> attributes)
child <item> id required string Unique identifier for this individual checkbox input.
child <item> label required string Display label for this checkbox option.
child <item> value required string The form value submitted when this checkbox is checked.
child <item> description optional string Additional descriptive text shown below the item label.
child <item> disabled optional boolean Set to "true" to disable this individual checkbox.
child <item> selected optional boolean Set to "true" to pre-check this option.

Basic Checkbox Group

Java

import com.google.common.collect.ImmutableList;

public class BasicCheckbox {
  public record Option(String id, String value, String label) {}

  public String name() {
    return "interests";
  }

  public String label() {
    return "Select your interests";
  }

  public ImmutableList<Option> items() {
    return ImmutableList.of(
      new Option("interest-reading", "reading", "Reading"),
      new Option("interest-sports", "sports", "Sports"),
      new Option("interest-music", "music", "Music"));
  }
}

Thymeleaf

<cf:checkbox
  th:name="${model.basicCheckbox.name()}"
  th:label="${model.basicCheckbox.label()}"
>
  <item
    th:each="item: ${model.basicCheckbox.items()}"
    th:id="${item.id()}"
    th:value="${item.value()}"
    th:label="${item.label()}"
  ></item>
</cf:checkbox>

Checkbox with Validation Error

Java

import com.google.common.collect.ImmutableList;

public class CheckboxWithError {
  public record Option(String id, String value, String label) {}

  public String name() {
    return "terms";
  }

  public String label() {
    return "Terms and conditions";
  }

  public String isValid() {
    return "false";
  }

  public String validationMessage() {
    return "You must accept at least one term to continue";
  }

  public ImmutableList<Option> items() {
    return ImmutableList.of(
      new Option("terms-service", "service", "I agree to the Terms of Service"),
      new Option("terms-privacy", "privacy", "I agree to the Privacy Policy"));
  }
}

Thymeleaf

<cf:checkbox
  th:name="${model.checkboxWithError.name()}"
  th:label="${model.checkboxWithError.label()}"
  th:is-valid="${model.checkboxWithError.isValid()}"
  th:validation-message="${model.checkboxWithError.validationMessage()}"
  required="true"
>
  <item
    th:each="item: ${model.checkboxWithError.items()}"
    th:id="${item.id()}"
    th:value="${item.value()}"
    th:label="${item.label()}"
  ></item>
</cf:checkbox>

Rendered HTML

<div class="usa-form-group usa-form-group--error">

    return "Available features";
  }

  public ImmutableList<Option> items() {
    return ImmutableList.of(
      new Option(
        "feature-basic",
        "basic",
        "Basic features",
        "checkbox-basic",
        "feature-basic-selected",
        false),
